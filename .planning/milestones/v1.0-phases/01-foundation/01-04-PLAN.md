  - If `b < 1024` return `b + ' B'`
  - If `b < 1024 * 1024` return `(b / 1024).toFixed(1) + ' KB'`

  `fmtPct(orig: number, opt: number): string`

  - Compute `saved = ((orig - opt) / orig) * 100`
  - If `saved === 0` return `''` (empty string — NOT em dash)
