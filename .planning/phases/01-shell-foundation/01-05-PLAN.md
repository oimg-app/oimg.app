       import { useState, useCallback } from 'react'

       export function DetailPanel() {
         const [splitPct, setSplitPct] = useState(50)

         const handleKeyDown = useCallback((e: React.KeyboardEvent) => {
           if (e.key === 'ArrowLeft') setSplitPct(p => Math.max(5, p - 5))
           if (e.key === 'ArrowRight') setSplitPct(p => Math.min(95, p + 5))
         }, [])

         return (
           <div
             className="relative h-full"
             style={{ background: 'var(--color-bg-0)' }}
           >
             {/* Phase 1 placeholder — compare view will be implemented in Phase 5 */}
             <div className="h-full flex items-center justify-center">

                 Select a file to compare
               </p>
             </div>

             {/* Split handle (keyboard accessible per UI-SPEC.md Interaction Contract) */}
             <div
