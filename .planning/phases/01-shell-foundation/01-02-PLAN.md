    Create src/tests/build.test.ts — bundle size smoke test (PERF-04).
    Use `spawnSync` (not `exec`/`execSync`) to invoke gzip — safer, no shell injection risk since all args are hardcoded file paths:

    ```typescript
    import { describe, it, expect } from 'vitest'
    import { readdirSync } from 'node:fs'
    import { resolve } from 'node:path'
    import { spawnSync } from 'node:child_process'

    // PERF-04: Initial route JS must be < 200 KB gzipped.
    // Skips gracefully when dist/ does not exist (run `npm run build` first).
    describe('Bundle size (PERF-04)', () => {
      it('initial JS bundle is under 200 KB gzipped', () => {
        const distAssetsDir = resolve(__dirname, '../../dist/assets')
        let jsFiles: string[] = []
        try {
          jsFiles = readdirSync(distAssetsDir).filter(
            f => f.endsWith('.js') && !f.includes('worker') && !f.includes('wasm')
          )
        } catch {
          console.warn('[bundle-size] dist/ not found — run npm run build first. Skipping.')
          return
        }

        let totalGzipBytes = 0
        for (const file of jsFiles) {
          const filePath = resolve(distAssetsDir, file)
          // spawnSync avoids shell injection: args are hardcoded paths, not user input
          const result = spawnSync('gzip', ['-c', filePath], { maxBuffer: 50 * 1024 * 1024 })
          if (result.status === 0 && result.stdout) {
            totalGzipBytes += result.stdout.length
          }
        }

        const totalKB = totalGzipBytes / 1024
        console.log(`[bundle-size] Initial JS gzip total: ${totalKB.toFixed(1)} KB`)
        // 200 KB budget per PERF-04
        expect(totalKB).toBeLessThan(200)
      })
    })
    ```
