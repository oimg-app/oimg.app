  interface BeforeInstallPromptEvent extends Event {
    prompt(): Promise<void>
    userChoice: Promise<{ outcome: 'accepted' | 'dismissed' }>
  }
  export const $installPrompt = atom<BeforeInstallPromptEvent | null>(null)
  export const $isInstalled = atom<boolean>(initial: matchMedia('(display-mode: standalone)').matches)

src/hooks/useInstallPrompt.ts:
  export function useInstallPrompt(): { canInstall: boolean; installed: boolean; promptInstall: () => Promise<boolean> }
  - useEffect: addEventListener 'beforeinstallprompt' (preventDefault + set atom),
    'appinstalled' (set $isInstalled true, clear $installPrompt); cleanup removes both
  - promptInstall: event.prompt() → await userChoice → clear atom → return outcome==='accepted'
  - canInstall = !!event && !installed

StatusBar (after the caps.offlineReady block):
