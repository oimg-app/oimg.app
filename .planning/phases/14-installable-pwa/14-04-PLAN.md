src/lib/register-sw.ts:
  import { registerSW } from 'virtual:pwa-register'
  import { toast } from 'sonner'
  import { setCaps } from '@/stores/runtime'
  import { probeCaps } from '@/lib/caps'
  export function bootstrapSW(): void
    - guard: if (!('serviceWorker' in navigator)) return
    - const updateSW = registerSW({
        onNeedRefresh() { toast('New version available', { action:{label:'Reload', onClick:()=>updateSW(true)}, duration:Infinity, id:'sw-update' }) }
        onOfflineReady() { setCaps({ ...probeCaps(), offlineReady: true }); toast.success('Ready to work offline', { duration:3000 }) }
      })

src/App.tsx useEffect (top-level):
  requestIdleCallback(() => void import('@/lib/register-sw').then(m => m.bootstrapSW()))
  fallback setTimeout 2000 when requestIdleCallback unavailable; cleanup cancels.
