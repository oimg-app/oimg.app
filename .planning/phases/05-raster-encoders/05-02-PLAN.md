**jpeg-adapter.ts:**
```typescript
// Phase 5 — JPEG (MozJPEG) encode adapter. OPT-04.
// Lazy-init: jpegMod null until first job. Static import path — Vite code-splits.
// D-10 (CONTEXT.md): all four raster adapters use lazy-init for consistency.
import type { AdapterMeta } from './types.ts'
import { AdapterError } from './types.ts'
import type { CodecSettingsJpeg } from '../types/index.ts'
import { buildJpegSettings } from './jpeg-config.ts'

type JpegModule = typeof import('@jsquash/jpeg')
let jpegMod: JpegModule | null = null
async function getJpeg(): Promise<JpegModule> {
  if (!jpegMod) jpegMod = await import('@jsquash/jpeg')
  return jpegMod
}

export async function run(
  input: ArrayBuffer,
  settings: unknown,
): Promise<{ output: ArrayBuffer; meta: AdapterMeta }> {
  const opts = buildJpegSettings({
    globalJpeg: settings as CodecSettingsJpeg,
    fileOverride: undefined, // merge happens in useBatchOrchestrate before dispatch
  })
  const { decode, encode } = await getJpeg()

  let decoded: ImageData
  try {
    decoded = await decode(input)
  } catch (err) {
    throw new AdapterError('jpeg', 'decode', err instanceof Error ? err.message : String(err))
  }

  let encoded: ArrayBuffer
  try {
    encoded = await encode(decoded, { quality: opts.quality, progressive: opts.progressive })
  } catch (err) {
    throw new AdapterError('jpeg', 'encode', err instanceof Error ? err.message : String(err))
  }

  return { output: encoded, meta: { codecVersion: 'jpeg@1.6.0' } }
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User-uploaded file → Worker ArrayBuffer | Untrusted binary data fed directly to WASM codec decode() |
| ICC chunk bytes → re-embedded in output | Extracted from user file; re-embedded verbatim — no parsing of ICC content |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-5-02-01 | Tampering | Malformed JPEG/WebP/AVIF bytes fed to jSquash decode() | mitigate | All three adapters wrap decode() in try/catch; throw AdapterError(format, 'decode', message); pool catches AdapterError and marks job as error — never crashes renderer |
| T-5-02-02 | Denial of Service | Extremely large input (>200 MB) exhausts WASM heap | mitigate | Phase 4 pool admission gate (byteEstimate) is already in place; adapters add no additional gate needed here |
| T-5-02-03 | Tampering | Malformed PNG iCCP chunk causes icc.ts to throw | mitigate | extractPngIcc() returns null on any parse error (not throw); png-adapter checks null before embedPngIcc() — gracefully skips ICC preservation on corrupt chunks |
| T-5-02-04 | Tampering | OxiPNG receives non-PNG bytes (e.g., if caller misconfigures) | mitigate | OxiPNG call is inside try/catch; throws AdapterError('png', 'process', ...) — worker propagates error to main thread, marks file as status='error' |
| T-5-02-05 | Information Disclosure | ICC profile data re-embedded in output contains user's device/camera profile | accept | ICC data is user's own asset metadata; user explicitly requested ICC preservation; no user of others' ICC data |
</threat_model>
